// ==UserScript==
// @name         IP + DateTime Overlay
// @namespace    http://tampermonkey.net/
// @version      1.3.1
// @description  Afișează overlay cu data, ora și IP public pe orice pagină (poziție implicită: centru jos)
// @author       Tu
// @match        *://*/*
// @grant        GM_xmlhttpRequest
// @grant        GM_addStyle
// @connect      ipify.org
// ==/UserScript==

(function() {
    'use strict';

    GM_addStyle(`
        #ip-overlay {
            position: fixed;
            left: 50%;
            bottom: 24px;
            transform: translateX(-50%);
            top: auto;
            right: auto;
            background: rgba(0,0,0,0.75);
            color: #ffffff;
            padding: 10px 14px;
            border-radius: 8px;
            font-family: system-ui, -apple-system, "Segoe UI", Roboto, "Helvetica Neue", Arial;
            font-size: 13px;
            z-index: 2147483647;
            box-shadow: 0 2px 8px rgba(0,0,0,0.5);
            backdrop-filter: blur(4px);
            cursor: grab;
            user-select: none;
            display: flex;
            gap: 14px;
            align-items: center;
            white-space: nowrap;
            transition: none;
        }
        #ip-overlay.dragging { cursor: grabbing; transform: none; }
        #ip-overlay .dt-text { color: #0b3d91; font-weight: 600; }
        #ip-overlay .time-text { color: #ffd400; font-weight: 700; }
        #ip-overlay .ip-text { color: #d32f2f; font-weight: 700; }
        #ip-overlay .ip-close {
            margin-left: 8px;
            background: transparent;
            border: none;
            color: #fff;
            font-weight: 700;
            cursor: pointer;
            font-size: 16px;
            line-height: 1;
        }
        @media (max-width:420px){
            #ip-overlay { font-size:12px; padding:6px 8px; gap:10px; left:50%; bottom:12px; transform:translateX(-50%); }
            #ip-overlay .ip-close { font-size:14px; }
        }
    `);

    const overlay = document.createElement('div');
    overlay.id = 'ip-overlay';
    overlay.setAttribute('role', 'status');
    overlay.setAttribute('aria-label', 'Data, ora și IP public');
    overlay.innerHTML = `<span class="dt-text">—</span><span class="time-text">—</span><span class="ip-text">IP: încărcare...</span><button class="ip-close" title="Închide" aria-label="Închide">×</button>`;
    document.documentElement.appendChild(overlay);

    const dtSpan = overlay.querySelector('.dt-text');
    const timeSpan = overlay.querySelector('.time-text');
    const ipSpan = overlay.querySelector('.ip-text');
    const closeBtn = overlay.querySelector('.ip-close');

    closeBtn.addEventListener('click', cleanupAndRemove);

    (function makeDraggable(el) {
        let isDown = false;
        let startX = 0, startY = 0, origX = 0, origY = 0;

        function ensurePosition() {
            if (!el.style.left && !el.style.top) {
                el.style.left = '50%';
                el.style.bottom = '24px';
                el.style.transform = 'translateX(-50%)';
            }
        }
        ensurePosition();

        const onPointerDown = (e) => {
            if (typeof e.button === 'number' && e.button !== 0) return;
            if (e.pointerId == null) return;
            try { if (el.setPointerCapture) el.setPointerCapture(e.pointerId); } catch (err) {}
            isDown = true;
            el.classList.add('dragging');
            startX = e.clientX;
            startY = e.clientY;
            const rect = el.getBoundingClientRect();
            const computed = getComputedStyle(el);
            if (computed.transform && computed.transform !== 'none') {
                el.style.left = rect.left + 'px';
                el.style.top = rect.top + 'px';
                el.style.transform = 'none';
                el.style.bottom = 'auto';
                el.style.right = 'auto';
            }
            const newRect = el.getBoundingClientRect();
            origX = newRect.left;
            origY = newRect.top;
            e.preventDefault();
        };
        const onPointerMove = (e) => {
            if (!isDown) return;
            const dx = e.clientX - startX;
            const dy = e.clientY - startY;
            el.style.left = (origX + dx) + 'px';
            el.style.top = (origY + dy) + 'px';
            el.style.right = 'auto';
            el.style.bottom = 'auto';
        };
        const onPointerUp = (e) => {
            if (!isDown) return;
            isDown = false;
            el.classList.remove('dragging');
            if (e.pointerId != null) {
                try { if (el.releasePointerCapture) el.releasePointerCapture(e.pointerId); } catch (err) {}
            }
        };

        el.addEventListener('pointerdown', onPointerDown);
        window.addEventListener('pointermove', onPointerMove);
        window.addEventListener('pointerup', onPointerUp);
    })(overlay);

    function fetchIP() {
        GM_xmlhttpRequest({
            method: "GET",
            url: "https://api.ipify.org?format=json",
            onload: function(res) {
                try {
                    const data = JSON.parse(res.responseText || '{}');
                    const ip = data.ip || 'necunoscut';
                    ipSpan.textContent = 'IP: ' + ip;
                } catch (e) {
                    ipSpan.textContent = 'IP: eroare';
                }
            },
            onerror: function() {
                ipSpan.textContent = 'IP: eroare';
            }
        });
    }

    function updateDateTime() {
        const now = new Date();
        const pad = (n) => String(n).padStart(2,'0');
        const formattedDate = `${pad(now.getDate())}-${pad(now.getMonth()+1)}-${now.getFullYear()}`;
        const formattedTime = `${pad(now.getHours())}:${pad(now.getMinutes())}:${pad(now.getSeconds())}`;
        dtSpan.textContent = formattedDate;
        timeSpan.textContent = formattedTime;
    }

    fetchIP();
    updateDateTime();

    const dtTimer = setInterval(updateDateTime, 1000);
    const ipTimer = setInterval(fetchIP, 5 * 60 * 1000);

    function cleanupAndRemove() {
        clearInterval(dtTimer);
        clearInterval(ipTimer);
        try { window.removeEventListener('beforeunload', cleanupAndRemove); } catch (e) {}
        if (overlay && overlay.parentNode) overlay.parentNode.removeChild(overlay);
    }

    window.addEventListener('beforeunload', cleanupAndRemove);
})();
